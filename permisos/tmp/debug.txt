SELECT proyectos.*,
`areas`.`area`
FROM
`proyectos`
LEFT OUTER JOIN `areas` ON (`proyectos`.`cod_area` = `areas`.`codigo`)
WHERE proyectos_status=1  and cod_tipo in ('a','b','c') and cod_area in ('6','5','7','3','10','4','1','8','2') and origen in ('CARTERA ACTUAL','PLADECO') and estado in ('En espera','En formulación de la iniciativa','Formulada','Postulada a financiamiento','Aprobada ','En licitación','En ejecución','Finalizada')
